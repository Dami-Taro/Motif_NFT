package src.main;

import java.nio.file.Path;
import java.nio.file.Paths;
import java.util.List;


import src.graph.Graph;
import src.io.GraphLoader;
import src.io.ResultWriter;
import src.motifMiner.TemporalMotifMiner;
import src.motifMiner.patterns.*;

public class Main {
    public static void main(String[] args) throws Exception {
        Graph g = GraphLoader.loadFromJson(Paths.get("collections/boredapeyachtclub.json"));
        //Graph g = GraphLoader.loadFromJson(Paths.get("collections/TestCaseInStar.json"));
        //Graph g = GraphLoader.loadFromJson(Paths.get("collections/TestCaseOneWayCouple.json"));
        

        System.out.println("Nodi: " + g.getNodes().size());
        System.out.println("Archi: " + g.getTotalEdges());

        //TemporalMotifMiner miner = new TemporalMotifMiner(g, 604800); // delta = 1 week = 604800 secondi
        TemporalMotifMiner miner = new TemporalMotifMiner(g, 350); // delta = 1 week = 604800 secondi
        
        List<OneWayCouple> oneWayCouplePatterns = miner.findOneWayCouples(2);
        //List<InStar> inStarPatterns = miner.findInStars(3);
        List<InStar> inStarPatterns = miner.findInStars(1);

        // ===== SCRITTURA SU FILE =====
        Path output = Paths.get("results/boredapeyachtclub/res_bayc_1week.txt");

        ResultWriter.writeGraphInfoToFile(g, 604800, output);

        ResultWriter.appendPatternCountsBySizeToFile("OneWayCouple", oneWayCouplePatterns, output);
        ResultWriter.appendPatternCountsBySizeToFile("InStar", inStarPatterns, output);

        System.out.println("✅ Risultati scritti in results/boredapeyachtclub/res_bayc_1week");
        
    }

}
